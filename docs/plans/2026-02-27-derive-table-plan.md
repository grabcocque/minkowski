# #[derive(Table)] Implementation Plan

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** Add a proc macro that turns a named struct into a compile-time schema — auto-implementing Bundle for spawning and generating typed accessors with cached column offsets that bypass HashMap lookups.

**Architecture:** `minkowski-derive` generates `Bundle` impl + `Table` trait impl + row reference types (`FooRef<'w>`, `FooMut<'w>`) from struct fields. `minkowski` defines the `Table` trait, `TableDescriptor` (cached archetype ID + column indices), and `World::query_table/query_table_mut` methods that use the descriptor to skip archetype matching.

**Tech Stack:** Rust 2021, `syn` + `quote` + `proc-macro2` for code generation, `minkowski` core crate.

```rust
use std::collections::HashMap;

use crate::bundle::Bundle;

/// Cached column layout for a Table type. Resolved once on first use.
pub struct TableDescriptor {
    pub archetype_id: ArchetypeId,
    /// Maps field index (0, 1, 2...) to archetype column index.
    pub col_indices: Vec<usize>,
}

/// A named struct that serves as both data container and schema declaration.
///
/// # Safety
/// Implementors (generated by `#[derive(Table)]`) must ensure that
/// `register` returns component IDs in the correct field order and
/// `FIELD_COUNT` matches the actual number of fields.
pub unsafe trait Table: Bundle + Sized {
    const FIELD_COUNT: usize;

    /// Register all field components. Returns ComponentIds in field declaration order.
    fn register(registry: &mut ComponentRegistry) -> Vec<ComponentId>;
}

/// Cache of resolved TableDescriptors, keyed by TypeId.
pub(crate) struct TableCache {
    descriptors: HashMap<TypeId, TableDescriptor>,
}

impl TableCache {
    pub fn new() -> Self {
        Self {
            descriptors: HashMap::new(),
        }
    }

    /// Get or create the descriptor for a Table type.
    pub fn get_or_create<T: Table>(
        &mut self,
        components: &mut ComponentRegistry,
        archetypes: &mut crate::storage::archetype::Archetypes,
    ) -> &TableDescriptor {
        let type_id = TypeId::of::<T>();
        self.descriptors.entry(type_id).or_insert_with(|| {
            // Register components in field order
            let field_ids = T::register(components);

            // Archetype uses sorted component IDs
            let mut sorted_ids = field_ids.clone();
            sorted_ids.sort_unstable();
            sorted_ids.dedup();
            assert_eq!(
                sorted_ids.len(),
                T::FIELD_COUNT,
                "duplicate component types in table"
            );

            let arch_id = archetypes.get_or_create(&sorted_ids, components);
            let archetype = &archetypes.archetypes[arch_id.0];

            // Map field index -> archetype column index
            let col_indices: Vec<usize> = field_ids
                .iter()
                .map(|comp_id| archetype.component_index[comp_id])
                .collect();

            TableDescriptor {
                archetype_id: arch_id,
                col_indices,
            }
        })
    }
}
```

```rust
/// Iterator over rows of a table's archetype, yielding raw column pointers per field.
pub struct TableIter<'w> {
    /// One pointer per field (in field declaration order), pointing to column start.
    col_ptrs: Vec<(*mut u8, usize)>, // (ptr, item_size)
    len: usize,
    current_row: usize,
    _marker: std::marker::PhantomData<&'w ()>,
}

impl<'w> TableIter<'w> {
    pub(crate) fn new(col_ptrs: Vec<(*mut u8, usize)>, len: usize) -> Self {
        Self {
            col_ptrs,
            len,
            current_row: 0,
            _marker: std::marker::PhantomData,
        }
    }
}

impl<'w> Iterator for TableIter<'w> {
    /// Yields a Vec of raw pointers, one per field in field order, pointing to the current row.
    type Item = Vec<*mut u8>;

    fn next(&mut self) -> Option<Self::Item> {
        if self.current_row >= self.len {
            return None;
        }
        let row = self.current_row;
        self.current_row += 1;
        Some(
            self.col_ptrs
                .iter()
                .map(|&(ptr, item_size)| unsafe { ptr.add(row * item_size) })
                .collect(),
        )
    }
}
```

```rust
    pub fn query_table<T: Table>(&mut self) -> TableIter<'_> {
        let desc = self.table_cache.get_or_create::<T>(
            &mut self.components,
            &mut self.archetypes,
        );
        let arch_id = desc.archetype_id;
        let col_indices = desc.col_indices.clone();

        let archetype = &self.archetypes.archetypes[arch_id.0];
        if archetype.is_empty() {
            return TableIter::new(vec![], 0);
        }

        let col_ptrs: Vec<(*mut u8, usize)> = col_indices
            .iter()
            .map(|&col_idx| unsafe {
                let ptr = archetype.columns[col_idx].get_ptr(0);
                let size = self.components.info(archetype.sorted_ids[col_idx]).layout.size();
                (ptr, size)
            })
            .collect();

        TableIter::new(col_ptrs, archetype.len())
    }

    pub fn query_table_mut<T: Table>(&mut self) -> TableIter<'_> {
        // Same implementation — mutability is enforced by &mut self
        self.query_table::<T>()
    }
```

```rust
use proc_macro::TokenStream;
use quote::quote;
use syn::{parse_macro_input, DeriveInput, Data, Fields};

/// Derive macro for Table types.
///
/// Generates:
/// - `unsafe impl Bundle` (component registration + put)
/// - `unsafe impl Table` (field count, register, descriptor)
/// - `FooRef<'w>` and `FooMut<'w>` typed row reference structs
#[proc_macro_derive(Table)]
pub fn derive_table(input: TokenStream) -> TokenStream {
    let input = parse_macro_input!(input as DeriveInput);
    let name = &input.ident;

    let fields = match &input.data {
        Data::Struct(data) => match &data.fields {
            Fields::Named(fields) => &fields.named,
            _ => panic!("#[derive(Table)] only supports structs with named fields"),
        },
        _ => panic!("#[derive(Table)] only supports structs"),
    };

    let field_names: Vec<_> = fields.iter().map(|f| f.ident.as_ref().unwrap()).collect();
    let field_types: Vec<_> = fields.iter().map(|f| &f.ty).collect();
    let field_count = field_names.len();

    // Generate Bundle impl
    let bundle_impl = quote! {
        unsafe impl ::minkowski::bundle::Bundle for #name {
            fn component_ids(
                registry: &mut ::minkowski::component::ComponentRegistry,
            ) -> Vec<::minkowski::component::ComponentId> {
                let mut ids = vec![
                    #(registry.register::<#field_types>()),*
                ];
                ids.sort_unstable();
                ids.dedup();
                assert_eq!(ids.len(), #field_count, "duplicate component types in table");
                ids
            }

            unsafe fn put(
                self,
                registry: &::minkowski::component::ComponentRegistry,
                func: &mut dyn FnMut(
                    ::minkowski::component::ComponentId,
                    *const u8,
                    std::alloc::Layout,
                ),
            ) {
                #(
                    let #field_names = std::mem::ManuallyDrop::new(self.#field_names);
                    func(
                        registry.id::<#field_types>().unwrap(),
                        &#field_names as *const std::mem::ManuallyDrop<#field_types>
                            as *const #field_types
                            as *const u8,
                        std::alloc::Layout::new::<#field_types>(),
                    );
                )*
            }
        }
    };

    // Generate Table impl
    let table_impl = quote! {
        unsafe impl ::minkowski::table::Table for #name {
            const FIELD_COUNT: usize = #field_count;

            fn register(
                registry: &mut ::minkowski::component::ComponentRegistry,
            ) -> Vec<::minkowski::component::ComponentId> {
                vec![
                    #(registry.register::<#field_types>()),*
                ]
            }
        }
    };

    // Generate Ref and Mut row types
    let ref_name = syn::Ident::new(&format!("{}Ref", name), name.span());
    let mut_name = syn::Ident::new(&format!("{}Mut", name), name.span());

    let ref_struct = quote! {
        pub struct #ref_name<'w> {
            #(pub #field_names: &'w #field_types,)*
        }
    };

    let mut_struct = quote! {
        pub struct #mut_name<'w> {
            #(pub #field_names: &'w mut #field_types,)*
        }
    };

    let expanded = quote! {
        #bundle_impl
        #table_impl
        #ref_struct
        #mut_struct
    };

    expanded.into()
}
```

```rust
pub struct TableTypedIter<'w, R: TableRow<'w>> {
    col_ptrs: Vec<(*mut u8, usize)>,
    len: usize,
    current_row: usize,
    _marker: std::marker::PhantomData<&'w R>,
}

impl<'w, R: TableRow<'w>> Iterator for TableTypedIter<'w, R> {
    type Item = R;

    fn next(&mut self) -> Option<R> {
        if self.current_row >= self.len {
            return None;
        }
        let row = self.current_row;
        self.current_row += 1;
        Some(unsafe { R::from_row(&self.col_ptrs, row) })
    }
}
```
